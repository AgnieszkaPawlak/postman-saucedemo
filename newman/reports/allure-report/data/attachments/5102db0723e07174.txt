// Status Code Validation
pm.test('Status code is 200', function() {
    pm.response.to.have.status(200);
});

// Content-Type Validation
pm.test('Content-Type is application/json', function() {
    pm.expect(pm.response.headers.get('Content-Type')).to.include('application/json');
});

// Response Time Check
pm.test('Response time is less than 2000ms', function() {
    pm.expect(pm.response.responseTime).to.be.below(2000);
});

// Cart Schema Validation
pm.test('Cart has required fields', function() {
    const cart = pm.response.json();
    pm.expect(cart).to.be.an('object');
    pm.expect(cart).to.have.property('id');
    pm.expect(cart).to.have.property('products');
    pm.expect(cart).to.have.property('total');
    pm.expect(cart.products).to.be.an('array');
    pm.expect(cart.total).to.be.a('number');
});
